🔀 Шартты конструкциялар

JavaScript-тегі шешім қабылдау құрылымдары

📋
if (егер)
if (жас >= 18) { console.log("Сіз кәмелетке толдыңыз!"); }

Шарт орындалса, код блогы іске қосылады. Ең қарапайым шартты конструкция.

🔄
if...else (егер...болмаса)
if (жас >= 18) { console.log("Кәмелетті"); } else { console.log("Кәмелетке толмаған"); }

Шарт орындалмаса, else блогы орындалады. Екі нұсқаның бірін таңдау үшін.

📊
else if (болмаса егер)
if (ұпай >= 90) { console.log("А"); } else if (ұпай >= 80) { console.log("В"); } else if (ұпай >= 70) { console.log("С"); } else { console.log("D"); }

Бірнеше шартты тексеру үшін. Біртіндеп шарттарды тексереді.

⚡
switch (ауыстыру)
switch (күн) { case "дүйсенбі": console.log("Апта басталды"); break; case "жұма": console.log("Апта соңы жақын"); break; default: console.log("Қалыпты күн"); }

Бір айнымалының көптеген мәндерін салыстыру үшін. else if-тің орнына.

⚡
Ternary оператор
// Қысқа жазу тәсілі let статус = (жас >= 18) ? "кәмелетті" : "кәмелетке толмаған"; // Немесе функцияларда function checkAge(age) { return age >= 18 ? "✅ Кіруге болады" : "❌ Кіруге болмайды"; }

Қысқа if...else. Бір жолда шартты тексеру үшін пайдаланылады.

🧠 Кешенді шартты логика

// Логикалық операторлармен let age = 25; let hasLicense = true; let hasInsurance = true; if (age >= 18 && hasLicense && hasInsurance) { console.log("Көлік айдауға болады"); } else if (age >= 18 && hasLicense) { console.log("Сақтандыру қажет"); } else if (age >= 18) { console.log("Құжаттар қажет"); } else { console.log("Жасыңыз жетпейді"); } // Баға категориясын анықтау let score = 95; let grade; if (score >= 97) { grade = "A+"; } else if (score >= 93) { grade = "A"; } else if (score >= 90) { grade = "A-"; } else if (score >= 87) { grade = "B+"; } else if (score >= 83) { grade = "B"; } else if (score >= 80) { grade = "B-"; } else if (score >= 70) { grade = "C"; } else { grade = "F"; } console.log(`Сіздің бағаңыз: ${grade}`);

⚡ Switch конструкциясы

// Аптаның күні function getDayMessage(day) { switch (day) { case "дүйсенбі": return "Апта басталды! 💪"; case "сейсенбі": case "сәрсенбі": case "бейсенбі": return "Жұмыс күні 📊"; case "жұма": return "Апта соңы жақын! 🎉"; case "сенбі": case "жексенбі": return "Демалыс күні! 🏖️"; default: return "Белгісіз күн"; } } // Калькулятор function calculate(a, operator, b) { switch (operator) { case "+": return a + b; case "-": return a - b; case "*": return a * b; case "/": return b !== 0 ? a / b : "Нөлге бөлуге болмайды"; default: return "Белгісіз оператор"; } } console.log(calculate(10, "+", 5)); // 15

🏗️ Ішкі шарттар (Nested conditions)

// Күрделі рөл тексеру function checkPermission(user) { if (user) { if (user.isActive) { if (user.role === "admin") { return "Барлық құқықтар"; } else if (user.role === "moderator") { return "Шектеулі құқықтар"; } else { return "Негізгі құқықтар"; } } else { return "Аккаунт өшірілген"; } } else { return "Пайдаланушы жоқ"; } } // Жақсартылған нұсқа (guard clauses) function checkPermissionBetter(user) { if (!user) return "Пайдаланушы жоқ"; if (!user.isActive) return "Аккаунт өшірілген"; switch (user.role) { case "admin": return "Барлық құқықтар"; case "moderator": return "Шектеулі құқықтар"; default: return "Негізгі құқықтар"; } } // Өнім санатын анықтау function categorizeProduct(product) { if (product.price > 0) { if (product.price <= 1000) { return product.category === "electronics" ? "Арзан электроника" : "Арзан тауар"; } else if (product.price <= 10000) { return "Орташа баға"; } else { return "Қымбат тауар"; } } else { return "Баға қате"; } }
